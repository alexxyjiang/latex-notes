\section*{数学符号格式表}
\centering
\begin{tabular}{ll}
\hline
$\mathbf{A}$ & 矩阵A \\
$\mathbf{A}_{m*n}$ & m*n维矩阵A \\
$\mathbf{a}_{ij}$ & 矩阵A的元素a（i行j列） \\
$\mathit{det}(\mathbf{A})$ & 矩阵A的行列式（值） \\
$\mathbf{A}^{T}$ & 矩阵A的转置 \\
$\mathit{adj}(\mathbf{A})$ & 矩阵A的伴随矩阵 \\
$\mathbf{\vec{a}}$ & 向量a \\
$\mathbf{\vec{a_{j}}}$ & 矩阵A的向量a（j列） \\
$\mathbf{I}$ & 单位矩阵 \\
$\mathbf{I}_{n}$ & n维单位矩阵 \\
\hline
\end{tabular}
